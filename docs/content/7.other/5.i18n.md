# 国际化
## 介绍

Dangoui 支持国际化，目前支持的语言有：

- 中文(zh-CN) 默认
- 英文(en-US)

## 使用

在 `App.vue` 中引入

```ts
// App.vue 中
import { provideLocale } from 'dangoui/locale/useLocale';
// 引入英文语言包
import enUS from 'dangoui/locale/en_US';
provideLocale(enUS);
```

在组件中使用

```vue
<template>
  <div>
    <button @click="changeToEnglish">Change to English</button>
  </div>
</template>

<script setup lang="ts">
// 在任何需要切换语言的组件中
import { useLocale } from 'dangoui/locale/useLocale'
import enUS from 'dangoui/locale/en_US';

const { setLocale } = useLocale()

// 切换到英文
function changeToEnglish() {
  setLocale(enUS)
}
</script>
```

#snippet
```vue
<template>
  <PreviewBlock title="支持展开收起">
    <DuButton @click="switchLanguage">切换 {{ currentLocale === 'zh_CN' ? 'English' : '中文' }}</DuButton>
    <DuTagsPanel
      :tags="tags"
      @add="handleAddClick"
      @remove="handleRemove"
      :collapse-count="5"
      can-toggle
    />
  </PreviewBlock>

</template>

<script setup lang="ts">
import { DuTagsPanel, DuButton } from 'dangoui'
import { useLocale, provideLocale } from 'dangoui/locale'
import enUS from 'dangoui/lang/en_US'
import zhCN from 'dangoui/lang/zh_CN'

const { setLocale, currentLocale } = useLocale()

onMounted(() => {
  provideLocale()
})

const tags = ['你好', 'Hello', '안녕하세요', 'Bonjour', 'こんにちは', 'Hallo'].map(text => ({ value: text, text, canRemove: text !== '你好' }))

function handleAddClick() {
  console.log('[demo] add click')
}

function handleRemove(item) {
  console.log('[demo] remove', item.value)
}

function switchLanguage() {
  console.log('[demo] change to english')
  setLocale(currentLocale.value === 'zh_CN' ? 'en_US' : 'zh_CN', currentLocale.value === 'zh_CN' ? enUS : zhCN)
}
</script>
```
