
```vue
          <DuInput
            placeholder="数量"
            :trim="true"
          />
        </DuFormItem>
        <DuFormItem label="限购" required>
          <DuInputNumber />
        </DuFormItem>
        <DuFormItem label="发货模式" required :show-border="false">
          <DuRadioGroup :value="1">
            <DuRadio label="直接物流" :value="1" />
            <DuRadio label="寄存柜" :value="2" />
            <DuRadio label="自提" :value="3" />

          发布

      </DuForm>

</template>
```
