```ts
import { DuSwitch } from '@frontend/dumpling'
```

## 示例

```vue
<template>
  <PreviewBlock title="简单">
    <DuSwitch v-model:on="isOn" />

  <PreviewBlock title="禁用">

  </PreviewBlock>
</template>

<script setup lang="ts">

import { DuSwitch } from '@frontend/dumpling'

const isOn = ref(false)
</script>
```
