# Dropdown 下拉选择器

:stage{type="alpha"}

下拉打开这个抽屉,喜欢的糖果都在里面🍬

```ts
import { DuDropdown } from 'dangoui'
```

## 使用注意

- `value` 为一个数组，即使只有一列

## 示例

::demo

## title: 单独使用

#snippet

```vue
<template>
  <PreviewBlock title="多维度筛选示例">
    <DuButton @click="handleVisible">打开筛选</DuButton>
    <DuDropdown
      :dimensions="dimensions"
      v-model:value="value"
      v-model:visible="visible"
      @confirm="handleConfirm"
    />
  </PreviewBlock>
</template>

<script setup lang="ts">
import { ref } from 'vue'
import { DuButton } from 'dangoui'
import DuDropdown from '/Users/zyr/Desktop/echo-project/dangoui/packages/dangoui/src/dropdown/Dropdown.vue'
const value = ref<Record<string, Record<string, string[]>>>({})
const visible = ref(false)

function handleVisible() {
  visible.value = true
}

function handleConfirm(selected: Record<string, Record<string, any[]>>) {
  console.log('选中的选项：', selected)
}

const dimensions = [
  {
    title: '品类',
    groups: [
      {
        title: '食品',
        multiple: true,
        options: [
          { label: '零食', value: 'snack' },
          { label: '饮料', value: 'drink' },
          { label: '生鲜', value: 'fresh' }
        ]
      },
      {
        title: '服装',
        multiple: false,
        options: [
          { label: '上衣', value: 'top' },
          { label: '裤子', value: 'pants' },
          { label: '鞋子', value: 'shoes', disabled: true }
        ]
      }
    ]
  },
  {
    title: '价格区间',
    groups: [
      {
        title: '价格',
        multiple: false,
        options: [
          { label: '0-50元', value: '0-50' },
          { label: '50-100元', value: '50-100' },
          { label: '100-200元', value: '100-200' },
          { label: '200元以上', value: '200+' }
        ]
      }
    ]
  }
]
</script>
```
