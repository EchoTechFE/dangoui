# Dropdown 下拉选择器

:stage{type="alpha"}

下拉打开这个抽屉,喜欢的糖果都在里面🍬

```ts
import { DuDropdown } from 'dangoui'
```

## 使用注意

- `value` 为嵌套对象结构，用于存储每个维度下各分组的选中值
- 组件支持单维度或多维度筛选
- 支持单选和多选模式

## 示例

title: 基础用法

最简单的单维度筛选

#snippet
```vue
<template>
  <PreviewBlock title="单维度筛选">
    <DuButton @click="handleShow" full size="large">选择价格区间</DuButton>
    <DuDropdown
      :dimensions="dimensions"
      v-model:value="value"
      v-model:visible="visible"
      @confirm="handleConfirm"
    />
    <!-- 结果展示 -->
    <div v-if="selectedItems.length > 0" class="result-display">
      <span class="result-label">已选择：</span>
      <DuTag 
        v-for="item in selectedItems" 
        :key="item.value"
        size="small"
        class="result-tag"
      >
        {{ item.label }}
      </DuTag>
      <DuButton type="text" size="small" @click="clearAll">清空</DuButton>
    </div>
  </PreviewBlock>
</template>

<script setup lang="ts">
import { ref } from 'vue'
import { DuButton, DuTag } from 'dangoui'
import DuDropdown from '/Users/zyr/Desktop/echo-project/dangoui/packages/dangoui/src/dropdown/Dropdown.vue'

const value = ref({})
const visible = ref(false)
const selectedItems = ref([])

function handleShow() {
  visible.value = true
}

function handleConfirm(selected) {
  console.log('选中的选项：', selected)
  // 提取所有选中的选项
  const items = []
  Object.values(selected).forEach(dimension => {
    if (Array.isArray(dimension)) {
      items.push(...dimension)
    } else {
      Object.values(dimension).forEach(group => {
        items.push(...group)
      })
    }
  })
  selectedItems.value = items
}

function clearAll() {
  selectedItems.value = []
  value.value = {}
}

const dimensions = [
  {
    label: '价格区间',
    value: 'price',
    multiple: false,
    options: [
      { label: '50元以下', value: '0-50' },
      { label: '50-100元', value: '50-100' },
      { label: '100-200元', value: '100-200' },
      { label: '200-500元', value: '200-500' },
      { label: '500元以上', value: '500+' }
    ]
  }
]
</script>

<style scoped>

  margin-top: 12px;
  font-size: 14px;
  flex-wrap: wrap;
}

  color: #666;
}
```

```vue
      v-model:value="value"
      v-model:visible="visible"
      @confirm="handleConfirm"
    />
    <!-- 分组展示结果 -->
    <div v-if="hasResults" class="complex-result">
      <div class="result-header">
        <h4>筛选结果</h4>
        <DuButton type="text" size="small" @click="clearAll" >清空全部</DuButton>
      </div>
      <div v-for="(dimension, dimKey) in selectedItems" :key="dimKey" class="result-dimension">
        <span class="dimension-label">{{ getDimensionTitle(dimKey) }}：</span>
        <!-- 处理扁平结构（直接 options） -->
        <div v-if="Array.isArray(dimension)" class="dimension-tags">
          <DuTag 
            v-for="item in dimension" 
            :key="item.value"
            size="small"
          >
            {{ item.label }}
          </DuTag>
        </div>
        <!-- 处理嵌套结构（groups） -->
        <div v-else class="dimension-groups">
          <div 
            v-for="(group, groupKey) in dimension" 
            :key="groupKey" 
            class="group-item"
          >
            <span class="group-label">{{ getGroupTitle(dimKey, groupKey) }}：</span>
            <div class="group-tags">
              <DuTag 
                v-for="item in group" 
                :key="item.value"
                size="small"
              >
                {{ item.label }}
              </DuTag>
            </div>
          </div>
        </div>
```

title: 自定义按钮文本
---

可以自定义取消和确认按钮的文本

#snippet
```vue
<template>
  <PreviewBlock title="自定义按钮文本">
    <DuButton @click="handleShow" full size="large">选择商品类型</DuButton>
    <DuDropdown
      :dimensions="dimensions"
      v-model:value="value"
      v-model:visible="visible"
      confirm-text="应用筛选"
      @confirm="handleConfirm"
    />
    <div v-if="selectedCount > 0" class="selection-info">
      已选择 {{ selectedCount }} 个选项
    </div>
  </PreviewBlock>
</template>

<script setup lang="ts">
import { ref, computed } from 'vue'
import { DuButton } from 'dangoui'
import DuDropdown from '/Users/zyr/Desktop/echo-project/dangoui/packages/dangoui/src/dropdown/Dropdown.vue'

const value = ref({})
const visible = ref(false)

const selectedCount = computed(() => {
  let count = 0
  Object.values(value.value).forEach(dimension => {
    Object.values(dimension).forEach(group => {
      count += Array.isArray(group) ? group.length : 0
    })
  })
  return count
})

function handleShow() {
  visible.value = true
}

function handleConfirm(selected) {
  console.log('应用筛选：', selected)
}

const dimensions = [
  {
    label: '商品类型',
    value: 'type',
    multiple: true,
    options: [
      { label: '新品', value: 'new' },
      { label: '热销', value: 'hot' },
      { label: '限量', value: 'limited' },
      { label: '预售', value: 'presale'}
    ]
  }
]
</script>

<style scoped>
.selection-info {
  margin-top: 12px;
  padding: 8px 12px;
  background: #f5f5f5;
  border-radius: 4px;
  font-size: 14px;
  color: #666;
}
</style>
```
